# 提示应用文件已经损坏

```bash
# 开启 任何来源
sudo spctl --master-disable

# 绕过公证
sudo xattr -rd com.apple.quarantine (拖入文件)

# 应用签名
# 安装 xcode
xcode-select --install
# 签名, 显示 "/文件位置 : replacing existing signature"
sudo codesign --force --deep --sign - (应用路径)
```
